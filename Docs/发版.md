# 2. 核心：--no-ff 合并（干净 + 可溯源）
git merge dev --no-ff -m "发版: v0.3.3 | 合并自 DEV 提交: $(git rev-parse dev)"

git tag v0.3.3
git push origin main --tags



极简操作步骤（仅操作 + 命令）
同步 DEV 最新代码
bash
运行
git checkout dev
git pull origin dev
基于 DEV 创建全新干净 main 分支
bash
运行
git checkout -b main
推送 main 到远程仓库
bash
运行
git push origin main
删除本地 / 远程所有旧标签（重置标签）
bash
运行
# 删除本地所有标签
git tag -d $(git tag)
# 删除远程所有标签
git push origin --delete $(git tag)
查看 DEV 所有版本标签（排序）
bash
运行
git checkout dev
git tag --sort=v:refname
逐版本合并到 main（对每个标签依次执行，以v0.0.1为例）
bash
运行
# 1. 切换到该版本
git checkout v0.0.1
# 2. 切换到main
git checkout main
# 3. --no-ff 合并（保留溯源）
git merge v0.0.1 --no-ff -m "发版 v0.0.1 | 源DEV提交: $(git rev-parse v0.0.1)"
# 4. main上打标签
git tag v0.0.1
# 5. 推送分支+标签
git push origin main
git push origin v0.0.1
